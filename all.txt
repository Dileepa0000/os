Task 2 - Process & Service Management

2.1 Basic Process Management

# Start a child process
bash

# Show process list with more information
ps aux | grep bash

# Show the Parent Process ID
echo $PPID

# Kill a process
kill [PID]

2.2 Advanced Process Management

# Create a pseudo process
yes "hello" &

# Set the priority
sudo renice -16 $(pgrep yes)

# Show the highest priority processes
ps -eo pid,ppid,ni,cmd --sort=-ni | head -10

# Gently kill the process
killall yes
2.3 Service Management

# Filter running services
systemctl list-units --type=service --state=running

# Check networking service status
systemctl status networking

# Disable and enable a service
sudo systemctl disable networking
sudo systemctl enable networking

# Stop a service and set it to start at boot
sudo systemctl stop networking
sudo systemctl enable networking

# Restart a service
sudo systemctl restart networking
Task 3 - Network Configuration

3.1 Basic Network Management


# Show IP configuration using 2 tools
ip addr show
ifconfig

# Shut down an interface
sudo ip link set eth0 down

# Set a static IP
sudo ip addr add 192.168.1.100/24 broadcast 192.168.1.255 dev eth0

# Change the MAC address
sudo ip link set dev eth0 address 00:11:22:33:44:55

# Bring the interface up
sudo ip link set eth0 up

# Change DNS
echo "nameserver 8.8.8.8" | sudo tee /etc/resolv.conf
echo "nameserver 1.1.1.1" | sudo tee -a /etc/resolv.conf
3.2 Advanced Network Configuration


# Use nmcli utilities
nmcli device show
nmcli connection show

# Disable a wired connection
nmcli connection down "Wired connection 1"

# Change DNS
nmcli connection modify "Wired connection 1" ipv4.dns "8.8.8.8"

# Enable the connection
nmcli connection up "Wired connection 1"
3.3 Text & File Management


# Pipe text to a file
echo "Linux is The Best" | sudo tee /tmp/printtext.txt

# Append a line using a command
echo "Pingu Rocks the Scene of Linux" | sudo tee -a /tmp/printtext.txt

# Rename a file
sudo mv /tmp/printtext.txt /tmp/editedtext.txt

# Directory operations
sudo mkdir /tmp/edited
sudo cp /tmp/editedtext.txt /tmp/edited/
sudo rm /tmp/editedtext.txt

# Create an empty file and move it
sudo touch /tmp/edited/empty.txt
sudo mv /tmp/edited/empty.txt /tmp/

# Remove a directory with its contents
sudo rm -r /tmp/edited
Task 4 - Permissions & CLI Processes

4.1 File Permissions


# Display file information
ls -la /etc/resolv.conf
stat /etc/resolv.conf

# Create a directory with spaces in the name
mkdir "/tmp/Long Live Linux"

# List directory contents with sizes in KB
ls -lh /etc/

# Find files starting with "serv"
find /etc/ -name "serv*"

# Create empty files
touch "/tmp/Long Live Linux/filemode1.txt" "/tmp/Long Live Linux/filemode2.txt"

# Change permissions
chmod u-r "/tmp/Long Live Linux/filemode1.txt"
chmod u+x "/tmp/Long Live Linux/filemode2.txt"
sudo chown root "/tmp/Long Live Linux/filemode2.txt"
sudo chgrp root "/tmp/Long Live Linux/filemode1.txt"
